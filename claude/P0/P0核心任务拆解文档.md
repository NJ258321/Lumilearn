## 错误处理标准模板

```typescript
try {
  // 业务逻辑
  const result = await prisma.xxx.create(...)

  res.status(201).json({
    success: true,
    data: result
  })
} catch (error: any) {
  console.error('Error operationName:', error)

  // Prisma 特定错误处理
  if (error.code === 'P2025') {
    return res.status(404).json({
      success: false,
      error: 'Resource not found'
    })
  }

  res.status(500).json({
    success: false,
    error: 'Failed to ...'
  })
}
```
