```typescript
// TimeMachine.tsx 核心算法
class TimeSyncEngine {
  private audioTime: number = 0
  private anchors: TimeAnchor[] = []

  // 1. 音频时间变化 -> 更新PPT和笔记
  onAudioTimeUpdate(time: number) {
    this.audioTime = time
    const anchor = this.findNearestAnchor(time)
    this.updatePPT(anchor?.pptPage)
    this.highlightNotes(anchor?.noteRange)
  }

  // 2. 点击笔记文字 -> 跳转音频和PPT
  onNoteClick(textPosition: number) {
    const anchor = this.findAnchorByNote(textPosition)
    this.audioPlayer.seek(anchor.timestamp)
    this.updatePPT(anchor.pptPage)
  }

  // 3. 时间锚点查找算法
  private findNearestAnchor(time: number): TimeAnchor {
    return this.anchors.reduce((prev, curr) => {
      return Math.abs(curr.timestamp - time) < Math.abs(prev.timestamp - time)
        ? curr : prev
    })
  }
}
```

```typescript
// 简化版逻辑
function detectWeakPoints(user: User): KnowledgePoint[] {
  const weakPoints = []
  for (const kp of user.knowledgePoints) {
    // 规则1：复习时间超过平均值
    if (kp.reviewTime > kp.avgReviewTime * 1.5) weakPoints.push(kp)
    // 规则2：错题率高
    if (kp.mistakeRate > 0.3) weakPoints.push(kp)
    // 规则3：图谱中有依赖但未学习
    if (kp.hasDependents && !kp.mastered) weakPoints.push(kp)
  }
  return weakPoints
}
```

```typescript
// 规划算法（简化版）
function generateDailyTasks(
  courses: Course[],
  examDate: Date,
  studyHoursPerDay: number
): DailyTask[] {
  const daysLeft = Math.ceil((examDate.getTime() - Date.now()) / (1000 * 60 * 60 * 24))
  const plan: DailyTask[] = []

  // 按紧迫度排序课程
  courses.sort((a, b) => a.examDate.getTime() - b.examDate.getTime())

  // 一轮：基础覆盖（40%时间）
  const round1Days = Math.floor(daysLeft * 0.4)
  for (let day = 0; day < round1Days; day++) {
    const tasks = courses.flatMap(c =>
      c.chapters.slice(day, day + 2).map(ch => ({
        type: 'review',
        courseId: c.id,
        chapterId: ch.id,
        estimatedTime: 30 // 分钟
      }))
    )
    plan.push({ day: day + 1, tasks, round: 1 })
  }

  // 二轮：重点突破（35%时间）
  const round2Days = Math.floor(daysLeft * 0.35)
  const weakPoints = findAllWeakPoints(courses)
  for (let day = 0; day < round2Days; day++) {
    const tasks = weakPoints.slice(day * 5, (day + 1) * 5).map(wp => ({
      type: 'focus_review',
      knowledgePointId: wp.id,
      estimatedTime: 20
    }))
    plan.push({ day: round1Days + day + 1, tasks, round: 2 })
  }

  // 三轮：模拟冲刺（25%时间）
  const round3Days = daysLeft - round1Days - round2Days
  const mistakes = getAllMistakes(courses)
  for (let day = 0; day < round3Days; day++) {
    const tasks = [
      { type: 'mock_exam', courseId: courses[0].id, estimatedTime: 60 },
      ...mistakes.slice(day * 3, (day + 1) * 3).map(m => ({
        type: 'mistake_review',
        mistakeId: m.id,
        estimatedTime: 15
      }))
    ]
    plan.push({ day: round1Days + round2Days + day + 1, tasks, round: 3 })
  }

  return plan
}

// 动态重排算法
function replanTasks(
  currentPlan: DailyTask[],
  completedTasks: Task[],
  today: number
): DailyTask[] {
  // 计算未完成的任务
  const pendingTasks = currentPlan
    .filter(d => d.day >= today)
    .flatMap(d => d.tasks.filter(t => !completedTasks.includes(t.id)))

  // 将未完成任务平摊到后续日期
  const remainingDays = currentPlan[currentPlan.length - 1].day - today + 1
  const tasksPerDay = Math.ceil(pendingTasks.length / remainingDays)

  const replanned = currentPlan.filter(d => d.day < today)
  for (let day = today; day < currentPlan.length; day++) {
    replanned.push({
      day,
      tasks: pendingTasks.slice((day - today) * tasksPerDay, (day - today + 1) * tasksPerDay),
      round: currentPlan[day - 1].round
    })
  }

  return replanned
}
```
